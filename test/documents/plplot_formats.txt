



<<fig_cap="sin(x) function."; label="sin_fun"; fig_pos="ht">>=
using PLplot
x = linspace(0, 2π, 200)
plot(x, sin(x), typ=:line)
@

<<echo=false; fig_cap="cos(x) function.">>=
plot(x, cos(x), typ=:line)
@

<<echo=false; label="cos2_fun">>=
plot(x, cos(2x), typ=:line)
@

<<term=true;fig_width=5;fig_ext=".png">>=
x = linspace(0, 2π, 200)
plot(x, sin(x))
y = 20
plot(x, cos(x))
@

<<term=true;fig_width=5;out_height="8cm">>=
x = linspace(0, 2π, 200)
plot(x, cos(x))
plot(x, sin(x))
@

<<out_width="15cm">>=
plot(x, sin(x))
lines(x, cos(x))
@
